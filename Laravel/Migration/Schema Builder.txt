常用的Schema方法
    建表
    Schema::create('表名',function(Blueprint $table){});

    針對指定的表進行維護
    Schema::table('表名',function(Blueprint $table){});

    重新命名資料表
    Schema::rename('原表名','新表名');

    刪除資料表
    Schema::drop('表名');

    存在時刪除
    Schema::dropIfExists('表名');
